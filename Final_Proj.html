Objective 3: Map the relationship between specific gene families and high-value agricultural traits like drought tolerance and disease resistance.

Creating trait labels and infer gene family

obj3 <- plant %>%
  mutate(
    trait_group = case_when(
      str_detect(str_to_lower(Application_Category), "drought tolerance") |
        str_detect(str_to_lower(Phenotype), "drought") ~ "Drought tolerance",

      str_detect(str_to_lower(Application_Category), "disease resistance") |
        str_detect(str_to_lower(Phenotype), "disease|blight|fungal|pathogen|viral|virus|resistance") ~ "Disease resistance",

      TRUE ~ NA_character_
    )
  ) %>%
  filter(!is.na(trait_group)) %>%
  mutate(
    Gene_Target = str_to_lower(Gene_Target),

    # keep only first listed target if multiple are present
    Gene_Target = str_split(Gene_Target, ",|;|/|\\|") %>% map_chr(1),
    Gene_Target = str_trim(Gene_Target),

    # remove common species prefixes
    Gene_Target_clean = str_remove(Gene_Target,
      "^(at|os|sl|zm|ta|hv|st|gh|gm|mt|rc|cs|le|pe|ib|pg|md|pa|vv|bn|bna|loc|loc_)[-_]?"
    ),

    gene_family = case_when(
      str_detect(Gene_Target_clean, "^myb") ~ "MYB",
      str_detect(Gene_Target_clean, "^nac") ~ "NAC",
      str_detect(Gene_Target_clean, "^wrky") ~ "WRKY",
      str_detect(Gene_Target_clean, "^dreb") ~ "DREB",
      str_detect(Gene_Target_clean, "^erf|^ap2") ~ "AP2/ERF",
      str_detect(Gene_Target_clean, "^bzip") ~ "bZIP",
      str_detect(Gene_Target_clean, "^mlo") ~ "MLO",
      str_detect(Gene_Target_clean, "^sweet") ~ "SWEET",
      str_detect(Gene_Target_clean, "r genes|r gene|r-gene") ~ "R genes",
      str_detect(Gene_Target_clean, "^pr") ~ "PR",
      TRUE ~ str_extract(str_to_upper(Gene_Target_clean), "^[A-Z]+")
    )
  ) %>%
  filter(!is.na(gene_family), gene_family != "")

Objective 4 (Quantitative Analysis):

Extract and analyze numerical data from Quant_Data text to evaluate whether CRISPR-edited plants demonstrate greater percentage improvements in yield compared to traditional transgenic plants.

define the technology groups

obj4_simple <- plant %>%
  mutate(
    tech_group = case_when(
      str_detect(str_to_lower(Mod_type), "crispr|cas9|cas12|base editing|prime editing") ~ "CRISPR",
      str_detect(str_to_lower(Mod_type), "transgenic|overexpression|agrobacterium|rnai|particle bombardment|t-dna|transformation") ~ "Transgenic",
      TRUE ~ NA_character_
    ),
    yield_related =
      str_detect(str_to_lower(Application_Category), "yield") |
      str_detect(str_to_lower(Phenotype), "yield") |
      str_detect(str_to_lower(Quant_Data), "yield"),

    percent_match = str_extract(Quant_Data, "\\d+\\.?\\d*\\s*%"),
    yield_pct = as.numeric(str_remove(percent_match, "%"))
  ) %>%
  filter(!is.na(tech_group), yield_related, !is.na(yield_pct))

summary

obj4_summary <- obj4_simple %>%
  group_by(tech_group) %>%
  summarise(
    n_values = n(),
    mean_pct = mean(yield_pct, na.rm = TRUE),
    median_pct = median(yield_pct, na.rm = TRUE),
    sd_pct = sd(yield_pct, na.rm = TRUE),
    min_pct = min(yield_pct, na.rm = TRUE),
    max_pct = max(yield_pct, na.rm = TRUE)
  )
